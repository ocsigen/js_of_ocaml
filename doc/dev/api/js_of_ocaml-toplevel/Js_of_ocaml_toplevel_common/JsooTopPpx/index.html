Module Js_of_ocaml_toplevel_common.JsooTopPpx

One can add a ppx rewriter to a toplevel by registering it

  let init () =
    Compiler_libs.Ast_mapper.register "js_of_ocaml" (fun _ -> Ppx_js.mapper)
